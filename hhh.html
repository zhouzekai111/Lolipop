body { font-family: Arial, sans-serif; background-color: #f4f4f4; display: flex; justify-content: center; align-items: center; height: 100vh; margin: 0; } .container { background-color: #fff; padding: 20px; border-radius: 8px; box-shadow: 0 0 10px rgba(0, 0, 0, 0.1); text-align: center; } input[type="text"] { padding: 10px; width: 80%; margin-bottom: 10px; border: 1px solid #ccc; border-radius: 4px; } button { padding: 10px 20px; background-color: #28a745; color: #fff; border: none; border-radius: 4px; cursor: pointer; } button:hover { background-color: #218838; } #poem { margin-top: 20px; font-style: italic; color: #333; } function generatePoem() { const keyword = document.getElementById('keyword').value; if (!keyword) { alert('请输入关键词'); return; } // 这里可以使用API或本地数据来生成诗歌 // 以下是一个简单的示例 const poems = [ `春风又绿江南岸，${keyword}何时照我还。`, `${keyword}如画，江山如此多娇。`, `举头望明月，低头思${keyword}。`, `${keyword}知时节，当春乃发生。`, `海内存知己，天涯若比${keyword}。` ]; const randomPoem = poems[Math.floor(Math.random() * poems.length)]; document.getElementById('poem').innerText = randomPoem; }
